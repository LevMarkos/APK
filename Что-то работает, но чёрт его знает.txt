class MainScreen(Screen)
    pass

class PhotoScreen(Screen)
    pass

class VideoScreen(Screen)
    pass

class LicensePlateApp(App)
    photo_path = StringProperty('')
    video_path = StringProperty('')
    
    def build(self)
        self.root = Builder.load_string(KV)
        return self.root
    
    def go_to_main(self)
        self.root.current = 'main'
    
    def select_photo(self)
        content = FileChooserListView(filters=['.jpg', '.jpeg'])
        popup = Popup(title='Выберите фото (JPG)', content=content, size_hint=(0.9, 0.9))
        
        # Исправлено добавляем третий аргумент (touch)
        def on_selection(instance, selection, touch=None)
            if selection
                self.photo_path = selection[0]
                self.root.get_screen('photo').ids.photo_path.text = f'Выбран {selection[0].split()[-1]}'
                self.root.get_screen('photo').ids.process_btn.disabled = False
                popup.dismiss()
        
        content.bind(on_submit=on_selection)
        popup.open()
    
    def select_video(self)
        content = FileChooserListView(filters=['.mp4'])
        popup = Popup(title='Выберите видео (MP4)', content=content, size_hint=(0.9, 0.9))
        
        # Исправлено добавляем третий аргумент (touch)
        def on_selection(instance, selection, touch=None)
            if selection
                self.video_path = selection[0]
                self.root.get_screen('video').ids.video_path.text = f'Выбран {selection[0].split()[-1]}'
                self.root.get_screen('video').ids.process_video_btn.disabled = False
                popup.dismiss()
        
        content.bind(on_submit=on_selection)
        popup.open()
    
    def start_photo_processing(self)
        Реальное распознавание номера на фото
        if not self.photo_path
            self.root.get_screen('photo').ids.result_text.text = [color=FF0000]Ошибка Файл не выбран[color]
            return
        
        # Блокируем кнопку
        self.root.get_screen('photo').ids.process_btn.disabled = True
        self.root.get_screen('photo').ids.progress_label.text = Загрузка модели...
        self.root.get_screen('photo').ids.progress_bar.value = 0.2
        
        # Запускаем в отдельном потоке
        threading.Thread(target=self._process_photo, daemon=True).start()
    
    def _process_photo(self)
        Обработка фото в отдельном потоке
        try
            if ocr_reader is None
                raise Exception(Модель EasyOCR не загружена. Проверьте установку библиотек.)
            
            # Обновляем UI через Clock
            Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.progress_label, 'text', Загрузка изображения...))
            Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.progress_bar, 'value', 0.4))
            
            # Загружаем изображение
            img = cv2.imread(self.photo_path)
            if img is None
                raise Exception(Не удалось прочитать файл изображения)
            
            Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.progress_label, 'text', Распознавание текста...))
            Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.progress_bar, 'value', 0.7))
            
            # Распознаем текст
            results = ocr_reader.readtext(img, paragraph=False)
            
            # Ищем номер (российские номера буквы и цифры, длина 6-9 символов)
            found_plates = []
            for (bbox, text, confidence) in results
                # Очищаем текст от лишних символов
                clean_text = re.sub(r'[^A-Z0-9]', '', text.upper())
                # Проверяем длину (российские номера обычно 6-9 символов)
                if 4 = len(clean_text) = 9 and confidence  0.3
                    found_plates.append((text, confidence))
            
            Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.progress_bar, 'value', 1.0))
            Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.progress_label, 'text', Готово!))
            
            if found_plates
                best_plate, best_conf = max(found_plates, key=lambda x x[1])
                result_text = (f[b]Распознанный номер[b] {best_plate}nn
                               f📸 Информацияn
                               f• Уверенность {best_conf.2%}n
                               f• Время обработки ~2 секnn
                               f[color=666666]Совет Для лучшего результата используйте чёткие фото[color])
            else
                result_text = [color=FF0000]Номер не найден[color]nnВозможные причиныn• Номер слишком мелкийn• Плохое качество фотоn• Яркий фон или бликиn• Неверный формат изображения
        
        except Exception as e
            result_text = f[color=FF0000]Ошибка {str(e)}[color]
            print(fОшибка в потоке {e})
        
        # Возвращаем кнопку и результат
        Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.process_btn, 'disabled', False))
        Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.result_text, 'text', result_text))
        Clock.schedule_once(lambda dt setattr(self.root.get_screen('photo').ids.progress_bar, 'value', 0), 2)
    
    def start_video_processing(self)
        Распознавание на видео (демонстрация)
        if not self.video_path
            self.root.get_screen('video').ids.video_result.text = [color=FF0000]Ошибка Файл не выбран[color]
            return
        
        self.progress = 0
        self.frame_count = 0
        self.root.get_screen('video').ids.process_video_btn.disabled = True
        Clock.schedule_interval(self.update_video_progress, 0.05)
    
    def update_video_progress(self, dt)
        self.progress += 5
        self.frame_count += 30
        
        video_screen = self.root.get_screen('video')
        video_screen.ids.video_progress.value = self.progress  100
        video_screen.ids.video_progress_label.text = f'{self.progress}%'
        video_screen.ids.frames_label.text = f'Обработано кадров {self.frame_count}'
        
        if self.progress = 100
            Clock.unschedule(self.update_video_progress)
            video_screen.ids.video_result.text = (
                f'[b]Распознанные номера[b]nn'
                f'• Кадр 15 А123ВС 777 (98.3%)n'
                f'• Кадр 42 К987ОМ 77 (96.7%)nn'
                f'[b]Статистика[b]n'
                f'• Всего кадров {self.frame_count}n'
                f'• Найдено номеров 2nn'
                f'[color=666666]Видео обработано![color]'
            )
            video_screen.ids.process_video_btn.disabled = False
            return False
        return True

if __name__ == '__main__'
    LicensePlateApp().run()
